	id = xalflavor_other.4
	title = xalflavor_other.4.t
	desc = xalflavor_other.4.d
	picture = FAMINE_eventPicture
	hidden = yes

	trigger = {
		owns = 5377
		had_global_flag	= {
			flag = rastrayvar_control_flag
			days = 1825
		}
	}

	mean_time_to_happen = {
		months = 1020
	}

	immediate = {
		hidden_effect = {
			clr_global_flag	= rastrayvar_control_flag
			random_country = {
				country_event = {
					id = xalflavor_other.5
					days = 30
				}
			}
			random_country = {
				limit = {
					any_owned_province = {
						OR = {
							region = north_xalia_region
							region = moxon_region
							area = glazia_area
						}
					}
				}
				country_event = {
					id = xalflavor_other.7
					days = 1
				}
			}
			every_country = {
				limit = {
					any_owned_province = {
						OR = {
							region = north_xalia_region
							region = moxon_region
							area = glazia_area
						}
					}
				}
				country_event = {
					id = xalflavor_other.6
				}
				if = {
					limit = {
						has_ruler = "Vergil"
					}
					country_event = {
						id = eastereggs.6
					}
				}
			}
		}
	}

	option = {
		name = xalflavor_other.4.a
		ai_chance = { factor = 1 }
	}
}
