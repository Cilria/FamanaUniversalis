country_event = {
	id = estal.1
	title = estal.1.t
	desc = estal.1.d
	picture = VILLAGE_BURNING_eventPicture
	
	trigger = {
        OR = {
			exists = HSH
			exists = HOF
		}
		NOT = {
			has_country_modifier = yom_raided_timer
		}
        NOT = {
			technology_group = antestal_yom_tech_gr
		}
		any_owned_province = {
			OR = {
				region = west_estal_region
				region = napora_region
				region = west_ordica_region
				area = area_149
				area = area_148
				area = area_136
				area = area_105 
				area = feyel_area
				area = unnamed_area_14
			}
        }
	}
	
	mean_time_to_happen = {
		months = 30
	}

	immediate = {
        hidden_effect = {
			add_country_modifier = {
                name = yom_raided_timer
                duration = 1825
                hidden = yes
            }
            random_owned_province = { 
				limit = {
					OR = {
					    region = west_estal_region
					    region = napora_region
					    region = west_ordica_region
					    area = area_149
					    area = area_148
					    area = area_136
					    area = area_105 
					    area = feyel_area
					    area = unnamed_area_14
					}
				}
				add_devastation = 15
				save_event_target_as = prov_raided_yoms
			}
		}
	}
	
	option = {
		name = "estal.1.a"
		ai_chance = {
			factor = 3
			modifier = {
				factor = 0
				num_of_loans = 1
			}
			modifier = {
				factor = 0
				NOT = { years_of_income	= 0.25 }
			}
		}
		add_years_of_income = -0.25
		goto = prov_raided_yoms
	}
	option = {
		name = "estal.1.b"
		ai_chance = { factor = 1 }
		event_target:prov_raided_yoms = {
		    cancel_construction = yes
            add_local_autonomy = 15
		    add_devastation = 10
		}
		goto = prov_raided_yoms
	}
}

country_event = { #lathvy conversion
	id = estal.2
	title = estal.2.t
	desc = estal.2.d
	picture = PRAYING_eventPicture
	
	trigger = {
		NOT = { has_country_flag = lathvy_convert_fired }
		NOT = { has_country_modifier = yomtanian_nation }
        OR = {
			religion_group = north_estali
			religion = animism
		}
		is_year = 350
		any_known_country = {
			religion_group = lathvy
		}
		capital_scope = {
			continent = estal
		}
		NOT = {
			government = native
		}
	}
	
	mean_time_to_happen = {
		months = 180
		modifier = {
			factor = 5
			capital_scope = {
				superregion = north_estal_superregion
			}
		}
	}

	immediate = {
		hidden_effect = {
			set_country_flag = lathvy_convert_fired
		}
	}
	
	option = { #Convert to lathvy
		name = "estal.2.a"
		ai_chance = { factor = 12 }
		add_stability = -2
		add_country_modifier = {
			name = "famana_conversion_zeal"
			duration = 3650
		}
		change_religion = lathvy
		capital_scope = {
			change_religion = lathvy
		}
	}

	option = { #Stay pagan
		name = "estal.2.b"
		ai_chance = { factor = 1 }
		add_country_modifier = {
			name = rejected_lathvy
			duration = 5475
		}
		change_government_reform_progress = -25
	}
}
