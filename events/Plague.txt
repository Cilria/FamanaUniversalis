#Maskrana - Measles equivalent
province_event = {
	id = plague.1
	title = plague.1.t
	desc = plague.1.d
	picture = PLAGUE_eventPicture
	goto = ROOT
	is_mtth_scaled_to_size = 1

	trigger = {
		development = 6
		has_port = yes
		NOT = {
			has_province_modifier = recent_maskrana_outbreak
		}
		NOT = {
						owner = {
				has_country_modifier = recent_disease
			}
		}
		NOT = {
			owner = {has_institution = printing_press} #edolyssa
		}
	}

	mean_time_to_happen = {
		months = 240
	}

	immediate = {
		hidden_effect = {
			
			add_province_modifier = {
				name = recent_maskrana_outbreak
				duration = 7300
			}
			owner = {
				add_country_modifier = {
					name = recent_disease
					duration = 3650
				}
			}
		}
	}
	
	option = { # Quarantine and Let die
		name = plague.1.a
		ai_chance = { factor = 3 }
		custom_tooltip = 5chanceSpread
		add_province_modifier = {
			name = totalQuarantine
			duration = 365
		}
		custom_tooltip = 2devlost
		hidden_effect = {
			random_list = {
				1 = {
					add_base_tax = -1
				}
				1 = {
					add_base_production = -1
				}
				1 = {
					add_base_manpower = -1
				}
			}
			random_list = {
				1 = {
					add_base_tax = -1
				}
				1 = {
					add_base_production = -1
				}
				1 = {
					add_base_manpower = -1
				}
			}
			random_list = {
				95 = {
					#do nothing
				}
				5 = {
					random_neighbor_province = {
						province_event = {
							id = plague.11
							days = 30
						}
						
					}
				}
			}
		}
		
	}
	option = { # Rely on terrible old timey doctors
		name = plague.1.b
		ai_chance = { factor = 2 }
		custom_tooltip = 25chanceSpread
		add_province_modifier = {
			name = local_doctors
			duration = 365
		}
		owner = {
			add_years_of_income = -0.25
		}
		custom_tooltip = 1devlost
		hidden_effect = {
			random_list = {
				1 = {
					add_base_tax = -1
				}
				1 = {
					add_base_production = -1
				}
				1 = {
					add_base_manpower = -1
				}
			}
			random_list = {
				75 = {
					#do nothing
				}
				25 = {
					random_neighbor_province = {
						province_event = {
							id = plague.11
							days = 30
						}
						
					}
				}
			}
		}
		
	}
	option = { # Do nothing
		name = plague.1.c
		ai_chance = { factor = 1 }
		custom_tooltip = 50chanceSpread
		add_province_modifier = {
			name = nothingDone
			duration = 365
		}
		hidden_effect = {
			random_list = {
				50 = {
					#do nothing
				}
				50 = {
					random_neighbor_province = {
						province_event = {
							id = plague.11
							days = 30
						}
						
					}
				}
			}

	trigger = {
		OR = {

		}
		OR = {
			trade_goods = grain #plants basically
			trade_goods = wine
			trade_goods = spices
			trade_goods = tea
			trade_goods = coffee
		}
		NOT = {
			has_province_modifier = recent_blight_outbreak
		}
		NOT = {
						owner = {
				has_country_modifier = recent_disease
			}
		}
		NOT = {
			has_global_flag = alymin_established
		}
	}

	mean_time_to_happen = {
		months = 240
	}

	immediate = {
		hidden_effect = {
			add_province_modifier = {
				name = recent_blight_outbreak
				duration = 7300
			}
			owner = {
				add_country_modifier = {
					name = recent_disease
					duration = 3650
				}
			}
		}
	}
	
	option = { # The harvest is ruined, and you cant do anything about it because you didnt study botany, boo hoo
		name = plague.5.a
		ai_chance = { factor = 3 }
		add_province_modifier = {
			name = noHarvest
			duration = 365
		}

			}
		}
	}
