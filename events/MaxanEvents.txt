namespace = maxan
#Bohoran culture shift
country_event = {
	id = maxan.1
	title = maxan.1.t
	desc = maxan.1.d
	picture = REFORM_eventPicture
	fire_only_once = yes

	trigger = {
		is_year = 280
		any_owned_province = {
			culture = Bohoran
		}
		OR = {
			tag = SKT
			NOT = {
				culture_group = Maxan
			}
		}
	}

	mean_time_to_happen = {
		months = 48
	}
	
	option = {
		name = maxan.1.a
		ai_chance = { factor = 8 }
		every_province = {
			limit = { culture = Bohoran }
			change_culture = Bohoran_E
		}
		if = {
			limit = { primary_culture = Bohoran }
			change_primary_culture = Bohoran_E
		}
		if = {
			limit = {
				accepted_culture = Bohoran
			}
			add_accepted_culture = Bohoran_E
			remove_accepted_culture	= Bohoran
		}
		hidden_effect = {
			every_country = {
				limit = {
					primary_culture = Bohoran
				}
				change_primary_culture = Bohoran_E
			}
			every_country = {
				limit = {
					accepted_culture = Bohoran
				}
				add_accepted_culture = Bohoran_E
				remove_accepted_culture	= Bohoran
			}
		}
	}
	
	option = {
		name = maxan.1.b
		ai_chance = { factor = 2 }
		add_dip_power = -50
		5361 = {
			add_trade_modifier = {
            	who = ROOT
            	duration = 3650
            	power = 25
            	key = maxan_focus
			}
		}
	}

}

country_event = {
	id = maxan.2
	title = maxan.2.t
	desc = maxan.2.d
	picture = REFORM_eventPicture
	fire_only_once = yes

	trigger = {
        NOT = {
            MOX = {
                has_country_flag = moxon_presence
            }
        }
		is_year = 280
		any_owned_province = {
			OR = {
                culture = West_Maxan
                culture = East_Maxan
            }
		}
		OR = {
            primary_culture = West_Maxan
            primary_culture = East_Maxan
        }
        NOT = {
			MOX = {
				any_owned_province = {
					OR = {
						region_for_scope_province= {
							region = stroktur_region
						}
						region_for_scope_province= {
							region = emastalia_region
						}
						region_for_scope_province= {
							region = west_emastalia_region
						}
					}
				}
			}
        }
	}

	mean_time_to_happen = {
		months = 48
	}
	
	option = {
		name = maxan.2.a
		ai_chance = { factor = 1 }
		every_province = {
			limit = { culture = West_Maxan }
			change_culture = West_Emastalian
		}
        every_province = {
			limit = { culture = East_Maxan }
			change_culture = East_Emastalian
		}

		if = {
			limit = {
				primary_culture = West_Maxan
			}
			change_primary_culture = West_Emastalian
			set_ruler_culture = West_Emastalian
		}

		if = {
			limit = {
				OR = {
					primary_culture = West_Maxan
					primary_culture = West_Emastalian
				}
				has_consort = yes
			}
			set_consort_culture = West_Emastalian
		}

			limit = {
				OR = {
					primary_culture = West_Maxan

			}

		}

				primary_culture = East_Maxan

				accepted_culture = West_Maxan
			}
			add_accepted_culture = West_Emastalian
			remove_accepted_culture	= West_Maxan
		}
        if = {
			limit = {
				accepted_culture = East_Maxan
			}
			add_accepted_culture = East_Emastalian
			remove_accepted_culture	= East_Maxan
		}
		hidden_effect = {
			every_country = {
				limit = {
					primary_culture = West_Maxan
				}
				set_ruler_culture = West_Emastalian
				change_primary_culture = West_Emastalian
			}
			every_country = {
				limit = {
					accepted_culture = West_Maxan
				}
				add_accepted_culture = West_Emastalian
				remove_accepted_culture	= West_Maxan
			}
            every_country = {
				limit = {
					primary_culture = East_Maxan
				}
				set_ruler_culture = East_Emastalian
				change_primary_culture = East_Emastalian
			}
			every_country = {
				limit = {
					accepted_culture = East_Maxan
				}
				add_accepted_culture = East_Emastalian
