namespace = maxan
#Bohoran culture shift
country_event = {
	id = maxan.1
	title = maxan.1.t
	desc = maxan.1.d
	picture = REFORM_eventPicture
	fire_only_once = yes

	trigger = {
		is_year = 280
		any_owned_province = {
			culture = Bohoran
		}
		OR = {
			tag = SKT
			NOT = {
				culture_group = Maxan
			}
		}
	}

	mean_time_to_happen = {
		months = 30
	}
	
	option = {
		name = maxan.1.a
		ai_chance = { factor = 8 }
		custom_tooltip = BohoranCultureGroupChange
		hidden_effect = {
			every_country = {
				limit = {
					primary_culture = Bohoran
				}
				change_primary_culture = Bohoran_E
			}
			every_country = {
				limit = {
					accepted_culture = Bohoran
				}
				add_accepted_culture = Bohoran_E
				remove_accepted_culture	= Bohoran
			}
			every_province = {
				limit = { culture = Bohoran }
				change_culture = Bohoran_E
			}
			if = {
				limit = { primary_culture = Bohoran }
				change_primary_culture = Bohoran_E
			}
			if = {
				limit = {
					accepted_culture = Bohoran
				}
				add_accepted_culture = Bohoran_E
				remove_accepted_culture	= Bohoran
			}
			if = {
				limit = {
					ruler_culture = Bohoran
				}
				set_ruler_culture = Bohoran_E
			}
			if = {
				limit = {
					consort_culture = Bohoran
				}
				set_consort_culture = Bohoran_E
			}
			if = {
				limit = {
					heir_culture = Bohoran
				}
				set_heir_culture = Bohoran_E
			}
		}
	}
	
	option = {
		name = maxan.1.b
		ai_chance = { factor = 2 }
		add_dip_power = -50
		5361 = {
			add_trade_modifier = {
            	who = ROOT
            	duration = 3650
            	power = 25
            	key = maxan_focus
			}
		}
	}

}

					has_country_flag = maxan_event_here

			limit = { culture = West_Maxan }

#Control event for 2
country_event = {
	id = maxan.4
	title = maxan.4.t
	desc = maxan.4.d
	picture = REFORM_eventPicture
	fire_only_once = yes
	hidden = yes

	trigger = {
		is_year = 270
        NOT = {
            MOX = {
                has_country_flag = moxon_presence
            }
        }
		any_owned_province = {
			OR = {
                culture = West_Maxan
                culture = East_Maxan
            }
		}
		OR = {
            primary_culture = West_Maxan
            primary_culture = East_Maxan
        }
        NOT = {
			MOX = {
				any_owned_province = {
					OR = {
						region_for_scope_province= {
							region = stroktur_region
						}
						region_for_scope_province= {
							region = emastalia_region
						}
						region_for_scope_province= {
							region = west_emastalia_region
						}
					}
				}
			}
        }
	}

	mean_time_to_happen = {
		months = 30
	}

	immediate = {
	    every_country = {
	        limit = {
				ai = no
	        	capital_scope = {
					region_for_scope_province = {
						OR = {
							region = emastalia_region
	        	        	region = west_emastalia_region
						}
					}
	        	}
			}
			country_event = {
	            id = maxan.2
	        }
			set_country_flag = maxan_event_here
	    }
	    random_country = {
	        limit = {
	            capital_scope = {
	                region_for_scope_province = {
						OR = {
							region = emastalia_region
	                    	region = west_emastalia_region
						}
					}
	            }
	        }
	        country_event = {
	            id = maxan.2
	        }
	    }
	}

	option = {
		name = maxan.4.a
		ai_chance = { factor = 1 }
	}
}
