country_event = {
	id = jasuaE.5
	hidden = yes
	fire_only_once = yes

	trigger = {
		is_year = 291
		owns = 5493
	}
	
	mean_time_to_happen = {
		months = 1
	}
	immediate = {
		set_global_flag	= eclipse_global_age_flag
		every_country = {
			limit = {
				NOT = {
					tag = SKT
				}
			}
			country_event = {
				id = jasuaE.6
			}
		}
		#First, check if Kalarga exists. Kalarga gets priority for Nafa.
		KLG = {
			if = {
				limit = {
					exists = yes
				}
				country_event = {
					id = jasuaE.14
				}
			}
		}
		#Then Stroktur. Republican Stroktur takes priority.
		SKT = {
			if = {
				limit = {
					NOT = { exists = KLG }
					NOT = { exists = NAF }
					has_country_flag = stroktur_republic
				}
				country_event = {
					id = jasuaE.7
				}
			}
			else_if = {
				limit = {
					NOT = { exists = KLG }
					NOT = { exists = NAF }
					any_owned_province = {
						culture = Kachik
					}
				}
				country_event = {
					id = jasuaE.13
				}
			}
			else = {
				country_event = {
					id = jasuaE.6
				}
			}
		}
		#Rachik and Yallwaka are next.
		RCK = {
			if = {
				limit = {
					NOT = { exists = KLG }
					NOT = { exists = NAF }
				}
				country_event = {
					id = jasuaE.15
				}
			}
		}
		YLL = {
			if = {
				limit = {
					NOT = { exists = KLG }
					NOT = { exists = RCK }
					NOT = { exists = NAF }
				}
				country_event = {
					id = jasuaE.15
				}
			}
		}
		#Finally, this means Nafa must have been conquered, so they will break free on their own.
		if = {
			limit = {
				NOT = { exists = KLG }
				NOT = { exists = RCK }
				NOT = { exists = NAF }
				NOT = { exists = YLL }
				NOT = { tag = SKT }
			}
			5488 = {
				owner = {
					country_event = {
						id = jasuaE.16
					}
				}
			}
		}
	}
	option = {
	}
}
