#eclipse controller
country_event = {
	id = jasuaE.5
	hidden = yes
	fire_only_once = yes

	trigger = {
		owns = 5493
		is_year = 291
		NOT = { is_year = 292 }
	}
	
	mean_time_to_happen = {
		months = 1
	}
	immediate = {
		set_global_flag	= eclipse_global_age_flag
		every_country = {
			limit = {
				NOT = {
					OR = {
						tag = SKT
						tag = RCK
						tag = NAF
						tag = KLG
						tag = YLL
					}
				}
			}
			country_event = {
				id = jasuaE.6
				days = 1
			}
		}
		#First, check if Kalarga exists. Kalarga gets priority for Nafa.
		KLG = {
			if = {
				limit = {
					exists = yes
				}
				country_event = {
					id = jasuaE.14
					days = 1
				}
			}
		}
		#Then Stroktur. Republican Stroktur takes priority.
		SKT = {
			if = {
				limit = {
					NOT = { exists = KLG }
					NOT = { exists = NAF }
					has_country_flag = stroktur_republic
				}
				country_event = {
					id = jasuaE.7
					days = 1
				}
			}
			else_if = {
				limit = {
					NOT = { exists = KLG }
					NOT = { exists = NAF }
					any_owned_province = {
						culture = Kachik
					}
				}
				country_event = {
					id = jasuaE.13
					days = 1
				}
			}
			else = {
				country_event = {
					id = jasuaE.6
					days = 1
				}
			}
		}
		#Rachik and Yallwaka are next.
		RCK = {
			if = {
				limit = {
					NOT = { exists = KLG }
					NOT = { exists = NAF }
				}
				country_event = {
					id = jasuaE.15
					days = 1
				}
			}
		}
		YLL = {
			if = {
				limit = {
					NOT = { exists = KLG }
					NOT = { exists = RCK }
					NOT = { exists = NAF }
				}
				country_event = {
					id = jasuaE.15
					days = 1
				}
			}
		}
		#Finally, this means Nafa must have been conquered, so they will break free on their own.
		if = {
			limit = {
				NOT = { exists = KLG }
				NOT = { exists = RCK }
				NOT = { exists = NAF }
				NOT = { exists = YLL }
				NOT = { tag = SKT }
			}
			5488 = {
				owner = {
					country_event = {
						id = jasuaE.16
						days = 1
					}
				}
			}
		}
	}
	option = {
	}
}
