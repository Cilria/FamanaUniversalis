# Rduma migration
country_event = {
    id = naravam.2
    title = naravam.2.t
    desc = naravam.2.desc
    picture = HORDE_ON_HORSEBACK_eventPicture
	fire_only_once = yes
    major = yes

    trigger = {
        AND = {
            tag = RDU
            is_year = 280
            NOT = {
                is_year = 350
            }
            liberty_desire = 20
            RDU = {
                AND = {
                    is_subject_of = ORC
                    owns_all_provinces = {
                        OR = {
                            area = nere_area
                            area = kletvar_area
                            area = area_57
                            area = feemk_area
                        }
                        NOT = {
                            any_owned_province = {
                                NOT = {
                                    area = nere_area
                                    area = kletvar_area
                                    area = area_57
                                    area = feemk_area
                                }
                            }
                        }
                    }
                }
            }
        }
    }

	mean_time_to_happen = {
		months = 360
	}

    option = {
        name = naravam.2.a
        ai_chance = { factor = 80 }
        trigger = {
            any_province = {
                OR = {
                    area = area_156
                    area = area_157
                }
                is_empty = yes
            }
        }
        goto = 6485
        hidden_effect = {
            if = {
                limit = {
                    exists = KLY
                    KLY = {
                        is_subject_of = ORC
                    }
                }
                feemk_area = {
                    cede_province = KLY
                    remove_core = RDU
                }
            }
            else = {
                feemk_area = {
                    cede_province = ORC
                    remove_core = RDU
                }
            }
            nere_area = {
                cede_province = ---
                remove_core = RDU
            }
            area_57 = {
                cede_province = ---
                remove_core = RDU
            }
	    6282 = {
                cede_province = ---
                remove_core = RDU
            }
	    6281 = {
                cede_province = ---
                remove_core = RDU
            }
           
	    6283 = { 
		     cede_province = KLV
		     add_core = KLV
		     remove_core = RDU 
	    }
            area_156 = {
                limit = {
				    is_empty = yes
			    }
                cede_province = RDU
                add_core = RDU
                change_culture = Rduma
                change_religion = naravami
            }
            area_157 = {
                limit = {
				    is_empty = yes
			    }
                cede_province = RDU
                add_core = RDU
                change_culture = Rduma
                change_religion = naravami
            }
            set_capital	= 6485
	    RDU = { change_religion = naravami
		    change_primary_culture = Rduma }
	    KLV = { change_religion = naravami
		    change_primary_culture = Rduma }
        }
        custom_tooltip = rdum_migrate_tooltip
        grant_independence = yes
        add_prestige = -10s
        add_country_modifier = {
            name = rdum_dev
            duration = 1825
        }
        add_adm_power = 150
        add_dip_power = 150
        add_mil_power = 150
    }

    option = {
        name = naravam.2.b
        ai_chance = { factor = 20 }
        add_country_modifier = {
            name = rdum_fight
            duration = 9125
        }
        add_prestige = 25
    }
}
