	desc = gaxalf.1.d
	picture = RELIGION_eventPicture

		is_year = 225

			is_year = 280

		}
	}

	}

		hidden_effect = {
			set_country_flag = has_had_arism
		}
	}

					superregion = ga_xal_superregion
  		  		}
   		   	}

			}
		}
		add_country_modifier = {
			name = antiarist_modifier
			duration = 3650
  		}
	}

		name = gaxalf.1.b
		ai_chance = { factor = 30 }

				}

   				who = ROOT

			}
		}

			name = arist_modifier

		}

	fire_only_once = yes

	trigger = {
		religion = delkian
	}
	mean_time_to_happen = {
		years = 8
	}

	option = {
		name = gaxalf.2.a
		ai_chance = { factor = 50 }
		add_estate_loyalty = {
			estate = estate_church
			loyalty = 5
		}
		random_owned_province = {
			limit = {
				NOT = {
					has_building = temple
				}

			}

		}
	}
	option = {
		name = gaxalf.2.b
		ai_chance = { factor = 50 }
		add_years_of_income = 1.25
	}
}
