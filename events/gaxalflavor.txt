#Rastrayvar control
country_event = {
	id = gaxalf.4
	title = gaxalf.4.t
	desc = gaxalf.4.d
	picture = FAMINE_eventPicture
	hidden = yes

	trigger = {
		OR = {
			had_global_flag	= {
				flag = rastrayvar_control_flag
				days = 1825
			}
			is_year = 217
		}
	}

	mean_time_to_happen = {
		months = 1000
	}

	immediate = {
		hidden_effect = {
			if = {
				limit = {
					has_global_flag = rastrayvar_control_flag
				}
				clr_global_flag	= rastrayvar_control_flag
				random_country = {
					country_event = {
						id = gaxalf.5
						days = 1
					}
				}
				random_country = {
					limit = {
						any_owned_province = {
							OR = {
								region = north_xalia_region
								region = moxon_region
								area = glazia_area
							}
						}
					}
					country_event = {
						id = gaxalf.7
						days = 1
					}
				}
			}
			else = {
				set_global_flag = rastrayvar_control_flag
			}
			every_country = {
				limit = {
					any_owned_province = {
						OR = {
							region = north_xalia_region
							region = moxon_region
							area = glazia_area
						}
					}
				}
				country_event = {
					id = gaxalf.6
				}
			}
		}
	}

	option = {
		name = gaxalf.4.a
		ai_chance = { factor = 1 }
	}
}
