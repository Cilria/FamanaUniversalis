#All scopes are country scope
#All parameters are optional
estate_church_kachik_religious_rights = {
	icon = privilege_promoto_dhimmi_nobles
	max_absolutism = -10
	loyalty = -0.05
	influence  = -0.10
    is_valid = {
		NOT = { tag = NAF }
		OR = {
			has_estate_privilege = estate_church_kachik_religious_rights
			AND = {
				tag = SKT
				NOT = { has_estate_privilege = estate_church_kachik_religious_rights }
			}
		}
		any_owned_province = {
			religion = animism
		}
		any_owned_province = {
			culture = Kachik
		}
	}
    can_select = {
		any_owned_province = {
			culture = Kachik
		}
	}
	#mechanics = {
	#	enables_cossack_regiments
	#}
	on_granted = {
		set_country_flag = kachikreligiousrights
		custom_tooltip = kachik_religious_rights_tip
	}
	on_granted_province = {
		if = {
			limit = {
				culture = Kachik
				religion = animism
			}
			add_province_modifier = {
				name = kachik_religious_autonomy
				duration = -1
			}
		}
	}
	on_revoked = {
		clr_country_flag = kachikreligiousrights
	}
	on_revoked_province = {
		if = {
			limit = {
				culture = Kachik
			}
			remove_province_modifier = kachik_religious_autonomy
		}
	}
	penalties = {
	}
	benefits = {
	}
    cooldown_years = 15
    ai_will_do = {
		factor = 1
		modifier = {
			factor = 10
			has_estate_privilege = estate_church_kachik_religious_rights
		}
	}
}
